
if NOT DEFINED IOTTREE_HOME set IOTTREE_HOME=%cd%

if "%1"=="install" goto install
if "%1"=="uninstall" goto uninstall
if "%1"=="start" goto start
if "%1"=="stop" goto stop

:install
wrapper.exe -i "%IOTTREE_HOME%/server/wrapper.conf"
goto finally


:uninstall
wrapper.exe -r "%IOTTREE_HOME%/server/wrapper.conf"
goto finally

:start
net start iottree_server
goto finally

:stop
net stop iottree_server
goto finally

:finally


ENDLOCAL
